cmake_minimum_required(VERSION 3.8)
project(LZ77AC)
set(CMAKE_C_STANDARD 99)
set(CMAKE_C_FLAGS "${CMAKE_C_FLAGS} -std=c99")
set(SOURCE_FILES src/main.c src/lz77/lz77.c src/lz77/lz77.h src/tools/file.c src/tools/file.h src/tools/fifo.c src/tools/fifo.h)
add_executable(LZ77AC ${SOURCE_FILES})
if (CMAKE_BUILD_TYPE MATCHES Debug)
    add_definitions(-DDEBUG=1)
    add_definitions(-DDEBUG_FIFO_VERBOSE_LOG=1)
    add_definitions(-DDEBUG_LZ77_LOG=1)
endif()
